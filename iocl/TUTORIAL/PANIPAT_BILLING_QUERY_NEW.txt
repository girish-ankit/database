#1>. VEHICLE NOT IN VEHICLE MASTER BUT PRESENT IN PANIPAT TOP BILLING REPORT: 10
>> SELECT * FROM BILLING_VEHICLE_TEST WHERE EVM_VEHICLE_NO NOT IN (SELECT EVM_VEHICLE_NO FROM ETRK_VEHICLE_MST)

#2>. VEHICLE NOT IN PANIPAT TOP BUT PRESENT IN BILLING REPORT:  82

>> SELECT  EVM_VEHICLE_NO FROM BILLING_VEHICLE_TEST WHERE EVM_VEHICLE_NO NOT IN (SELECT EVM_VEHICLE_NO FROM ETRK_VEHICLE_MST WHERE EVM_SUPPLY_LOC_ID = 14);

#3>. TOATAL NUMBER OF VEHICELS COMES FROM OTHER TERMINAL: 82-10 = 72


>> SELECT LM.ESLM_SUPPLY_LOC_STATE, LM.ESLM_SUPPLY_LOC_NAME, VM.* FROM ETRK_VEHICLE_MST VM left JOIN ETRK_SUPPLY_LOC_MST LM ON VM.EVM_SUPPLY_LOC_ID= LM.ESLM_SUPPLY_LOC_ID
WHERE VM.EVM_VEHICLE_NO IN (SELECT EVM_VEHICLE_NO FROM BILLING_VEHICLE_TEST WHERE EVM_VEHICLE_NO NOT IN (SELECT EVM_VEHICLE_NO FROM ETRK_VEHICLE_MST WHERE EVM_SUPPLY_LOC_ID = 14)) order by EVM_VEHICLE_NO;

#4>. TOATL NUMBER OF VEHICLE NOT USED BY PANIPAT TOP DEPOT (MISSING IN REPORT)

 >> SELECT * FROM ETRK_VEHICLE_MST WHERE EVM_VEHICLE_NO NOT IN (SELECT EVM_VEHICLE_NO FROM BILLING_VEHICLE_TEST)
AND EVM_SUPPLY_LOC_ID = 14;

#5>. VEHICLE USED BY PANIPAT DEPOT FROM OTHER DEPOT:

SELECT LM.ESLM_SUPPLY_LOC_STATE, LM.ESLM_SUPPLY_LOC_NAME, VM.* FROM ETRK_VEHICLE_MST VM left JOIN ETRK_SUPPLY_LOC_MST LM ON VM.EVM_SUPPLY_LOC_ID= LM.ESLM_SUPPLY_LOC_ID
WHERE VM.EVM_VEHICLE_NO IN (SELECT EVM_VEHICLE_NO FROM BILLING_VEHICLE_TEST WHERE EVM_VEHICLE_NO NOT IN (SELECT EVM_VEHICLE_NO FROM ETRK_VEHICLE_MST WHERE EVM_SUPPLY_LOC_ID = 14)) order by EVM_VEHICLE_NO;

#6>. INVOICE DATA ON PANIPAT DEPOT FROM ANOTHER DEPOT:

>> SELECT RM.IRM_DEPOT_NAME, RM.IRM_CONSUMER_NAME, ID.* FROM IOCL_INVOICE_DATA ID JOIN IOCL_ROUTE_MST RM
ON ID.IID_ROUTE_ID = RM.IRM_ROUTE_ID
WHERE ID.IID_INVOICE_DATE > STR_TO_DATE('31/9/2015 00:00:00', '%d/%m/%Y %H:%i:%s')
AND ID.IID_INVOICE_DATE < STR_TO_DATE('1/11/2015 00:00:00', '%d/%m/%Y %H:%i:%s')
AND UPPER(RM.IRM_DEPOT_NAME) = UPPER('Panipat Top')
AND IID_VEHICLE_NO IN (SELECT VM.EVM_VEHICLE_NO FROM ETRK_VEHICLE_MST VM JOIN ETRK_SUPPLY_LOC_MST LM ON VM.EVM_SUPPLY_LOC_ID= LM.ESLM_SUPPLY_LOC_ID
WHERE VM.EVM_VEHICLE_NO IN (SELECT EVM_VEHICLE_NO FROM BILLING_VEHICLE_TEST WHERE EVM_VEHICLE_NO NOT IN (SELECT EVM_VEHICLE_NO FROM ETRK_VEHICLE_MST WHERE EVM_SUPPLY_LOC_ID = 14))) order by IID_VEHICLE_NO;


#7>. OHTER TERMIANL VEHICLE INFO THAT ARE PRESENT IN PANIPAT BILL BUT NOT IN INV0ICE DATA:

>> select * from (
SELECT LM.ESLM_SUPPLY_LOC_STATE, LM.ESLM_SUPPLY_LOC_NAME, VM.* FROM ETRK_VEHICLE_MST VM left JOIN ETRK_SUPPLY_LOC_MST LM ON VM.EVM_SUPPLY_LOC_ID= LM.ESLM_SUPPLY_LOC_ID
WHERE VM.EVM_VEHICLE_NO IN (SELECT EVM_VEHICLE_NO FROM BILLING_VEHICLE_TEST WHERE EVM_VEHICLE_NO NOT IN (SELECT EVM_VEHICLE_NO FROM ETRK_VEHICLE_MST WHERE EVM_SUPPLY_LOC_ID = 14))
) a where a.EVM_VEHICLE_NO not in (
SELECT DISTINCT ID.IID_VEHICLE_NO FROM IOCL_INVOICE_DATA ID JOIN IOCL_ROUTE_MST RM
ON ID.IID_ROUTE_ID = RM.IRM_ROUTE_ID
WHERE ID.IID_INVOICE_DATE > STR_TO_DATE('31/9/2015 00:00:00', '%d/%m/%Y %H:%i:%s')
AND ID.IID_INVOICE_DATE < STR_TO_DATE('1/11/2015 00:00:00', '%d/%m/%Y %H:%i:%s')
AND UPPER(RM.IRM_DEPOT_NAME) = UPPER('Panipat Top')
AND IID_VEHICLE_NO IN (SELECT VM.EVM_VEHICLE_NO FROM ETRK_VEHICLE_MST VM JOIN ETRK_SUPPLY_LOC_MST LM ON VM.EVM_SUPPLY_LOC_ID= LM.ESLM_SUPPLY_LOC_ID
WHERE VM.EVM_VEHICLE_NO IN (SELECT EVM_VEHICLE_NO FROM BILLING_VEHICLE_TEST WHERE EVM_VEHICLE_NO NOT IN (SELECT EVM_VEHICLE_NO FROM ETRK_VEHICLE_MST WHERE EVM_SUPPLY_LOC_ID = 14))) order by IID_VEHICLE_NO
);
