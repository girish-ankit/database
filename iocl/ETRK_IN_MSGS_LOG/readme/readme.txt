select DATE_FORMAT(EIM_MESSAGE_CREATED_TIME, '%d/%m/%Y %h:%i %p') as Date_minute, COUNT(EIM_SLNO) as rows from ETRK_IN_MSGS 
WHERE EIM_MESSAGE_CREATED_TIME BETWEEN STR_TO_DATE('7/12/2015 00:00:00', '%d/%m/%Y %H:%i:%s') AND STR_TO_DATE('7/12/2015 23:59:59', '%d/%m/%Y %H:%i:%s')
GROUP BY HOUR(EIM_MESSAGE_CREATED_TIME);
