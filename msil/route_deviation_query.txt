SELECT
  DISTINCT IRD_VEH_ID
 
FROM IOCL_ROUTE_DEVIATIONS_NEW
WHERE IRD_DEV_DUR_IN_MINS>=5
AND IRD_IS_DONE =0 AND IRD_DEV_END_AT >= DATE_ADD(CURDATE(), INTERVAL -1 DAY) AND IRD_DEV_END_AT < DATE_ADD(CURDATE(), INTERVAL -0 DAY) ORDER BY IRD_ID DESC;


SELECT DISTINCT IAC_IID_VEHICLE_NO FROM IOCL_ALLIDLE_CLOSEDTRIPS WHERE IAC_IID_TRIP_ACTUAL_END_TIME >= DATE_ADD(CURDATE(), INTERVAL -1 DAY) AND IAC_IID_TRIP_ACTUAL_END_TIME < DATE_ADD(CURDATE(), INTERVAL -0 DAY);
SELECT DISTINCT IAC_IID_VEHICLE_NO FROM IOCL_ALLIDLE_CLOSEDTRIPS WHERE IAC_IID_TRIP_FORCE_CLOSED_TIME >= DATE_ADD(CURDATE(), INTERVAL -1 DAY) AND IAC_IID_TRIP_FORCE_CLOSED_TIME < DATE_ADD(CURDATE(), INTERVAL -0 DAY);
