#1>.PROCEDURE NAME: MSIL_JOURNEY_LOG_HTP

#2>. TABLE NAME: MSIL_JOURNEY_LOG_HTP_NEW


#3>.HOW TO CALL PROCEDURE: 

DECLARE
  VEHICLENO VARCHAR2(200);
  FROMDATE VARCHAR2(200);
  TODATE VARCHAR2(200);
BEGIN
-- date formate must be in DD/MM/YYYY
-- DATE RANGE IS LIKE  TMP_ERT >= FROMDATE AND TMP_ERT < TODATE
  VEHICLENO := 'HR55K4695';
  FROMDATE := '12/05/2015';
  TODATE := '26/05/2015';

  MSIL_JOURNEY_LOG_HTP(
    VEHICLENO => VEHICLENO,
    FROMDATE => FROMDATE,
    TODATE => TODATE
  );
--rollback; 
END;




#4>. RUN FOLLWOING QUERY TO GET DATA

SELECT * FROM MSIL_JOURNEY_LOG_HTP_NEW ORDER BY TMP_ERT DESC;
